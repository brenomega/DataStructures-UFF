TLSE *insertStart(TLSE *current, int info) {
	TLSE *new = (TLSE*) malloc(sizeof(TLSE));
	new->info = info;
	new->next = current;
	return new;
}

TLSE *insertMiddle(TLSE *current, int info, int position) {
	if (!current || position == 0) {
		TLSE *new = insertStart(current, info);
		return new;
	}

	if (position > 1) {
		current->next = insertMiddle(current->next, info, position - 1);
	}

	return current;
}
